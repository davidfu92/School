def objective(w, x, y, lam):
    n, d = x.shape
    yx = np.multiply(y, x) # n-by-d matrix
    yxw = np.dot(yx, w) # n-by-1 matrix
    vec1 = np.exp(-yxw) # n-by-1 matrix
    vec2 = np.log(1 + vec1) # n-by-1 matrix
    loss = np.mean(vec2) # scalar
    reg = lam / 2 * np.sum(w * w) # scalar
    return loss + reg

def gradient(w, x, y, lam):
    n, d = x.shape
    yx = np.multiply(y, x) # n-by-d matrix
    yxw = np.dot(yx, w) # n-by-1 matrix
    vec1 = np.exp(yxw) # n-by-1 matrix
    vec2 = np.divide(yx, 1+vec1) # n-by-d matrix
    vec3 = -np.mean(vec2, axis=0).reshape(d, 1) # d-by-1 matrix
    g = vec3 + lam * w
    return g

def grad_descent(x, y, lam, stepsize, max_iter=100, w=None):
    n, d = x.shape
    objvals = np.zeros(max_iter) # store the objective values
    if w is None:
        w = np.zeros((d, 1)) # zero initialization
    
    for t in range(max_iter):
        objval = objective(w, x, y, lam)
        objvals[t] = objval
        print('Objective value at t=' + str(t) + ' is ' + str(objval))
        g = gradient(w, x, y, lam)
        w -= stepsize * g
    
    return w, objvals

class Slave:
    def __init__(self,x, y):
        self.x = x
        self.y = y
        self.s = x.shape[0]
        self.d = x.shape[1]
        self.w = np.zeros((d, 1))
        
    def set_weight(self, w):
        self.w = w    
        
    def loss(self, lam): 
        return numpy.log(1 + numpy.exp(-yxw))
    
    def grad_descent(self, q, alpha, lam): 
        objvals = np.zeros(q)
        w = self.w
        for t in range(q):
            objval = objective_func(w, self.x, self.y, lam)
            g = gradient(w, self.x, self.y, lam)
            w -= alpha * g
            objvals[t] = objval
            alpha *= 0.9
        g = self.w - w
        return g, objval
    
    def sgd(w, xi, yi, lam):
        d = xi.shape[0]
        yx = yi * xi # 1-by-d matrix
        yxw = float(numpy.dot(yx, w)) # scalar

        # calculate objective function Q_i
        loss = numpy.log(1 + numpy.exp(-yxw)) # scalar
        reg = lam / 2 * numpy.sum(w * w) # scalar
        obj = loss + reg

        # calculate stochastic gradient
        g_loss = -yx.T / (1 + numpy.exp(yxw)) # d-by-1 matrix
        g = g_loss + lam * w # d-by-1 matrix

        return obj, g

class Master:
    def __init__(self, m, n, d):
        self.m = m
        self.n = n
        self.d = d
        self.w = np.zeros((d, 1))
        self.g = np.zeros((d, 1)) 
        self.v = np.zeros((d, 1))
        self.loss = 0
        self.obj = 0
        
    def broadcast_weight(self): 
        return self.w
    
    def collect(self, grads, losses): 
        self.g = np.zeros((self.d, 1)) 
        self.loss=0
        for k in range(self.m):
            self.g += grads[k]
            self.loss += losses[k]
    
    def gradient(self, lam):
        self.g = self.g / self.n 
    
    def objective_func(self, lam):
        yx = np.multiply(self.y, self.x)
        yxw = np.dot(yx, w)
        vec1 = np.exp(-yxw)
        vec2 = np.log(1 + vec1)
        loss = np.mean(vec2)
        reg = lam / 2 * np.sum(w * w)
        return loss + reg
    
    def net_gd(self, alpha, beta):
        self.v *= beta
        self.v += self.g
        self.w -= alpha * self.v

def create_master_slaves(m,x, y): 
    n, d = x.shape
    s = math.floor(n / m) 
    master = Master(m, n, d) 
    slaves = []

    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        worker = Slave(x[indices, :], y[indices, :]) 
        slaves.append(worker)
    return master, slaves
